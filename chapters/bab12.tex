\subsection{Alur Penggunaan Utama}

\subsubsection{Alur untuk Customer}
\begin{enumerate}[leftmargin=*]
    \item Buka aplikasi dan login dengan akun customer
    \item Jelajahi marketplace untuk mencari makanan yang diinginkan
    \item Klik produk untuk melihat detail dan \textit{Food Trust Index}
    \item Tambahkan ke keranjang dan lakukan checkout
    \item Pilih metode pembayaran dan konfirmasi pesanan
    \item Pantau status pesanan melalui fitur tracking
    \item Berikan rating dan ulasan setelah makanan diterima
\end{enumerate}

\subsubsection{Alur untuk UMKM}
\begin{enumerate}[leftmargin=*]
    \item Login dengan akun UMKM
    \item Tambahkan produk baru melalui menu ``Kelola Produk''
    \item Isi form \textit{Food Eligibility Assessment}
    \item Lihat \textit{Food Eligibility Score} yang dihasilkan sistem
    \item Atur harga (manual atau ikuti saran \textit{Dynamic Pricing})
    \item Publikasikan produk ke marketplace
    \item Proses pesanan yang masuk dan update status
\end{enumerate}

\subsubsection{Alur untuk Admin}
\begin{enumerate}[leftmargin=*]
    \item Login dengan akun admin
    \item Akses dashboard untuk melihat ringkasan aktivitas
    \item Kelola akun pengguna melalui menu ``Manajemen User''
    \item Monitor produk yang dipublikasikan
    \item Verifikasi permintaan donasi
    \item Tangani laporan dari pengguna
\end{enumerate}
